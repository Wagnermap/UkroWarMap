Настройка карты (Спутник вместо черного фона) // Используем ESRI Satellite — он четче, чем Sentinel-2 и не имеет его водяных знаков var satellite = L.tileLayer('https://server.arcgisonline.com/ArcGIS/rest/services/World_Imagery/MapServer/tile/{y}/{x}/{z}', { attribution: 'Imagery © Esri' }).addTo(map); // 2. Функция автоматического выбора цвета function getLayerStyle(path, feature) { // Окопы — всегда Желтые if (path.includes('trenches.json')) { return { color: "#FFFF00", weight: 3, opacity: 0.9 }; } // Всё из папки Russia — Красное if (path.includes('/russia/')) { return { color: "#FF0000", weight: 2, opacity: 0.8 }; } // Всё из папки Ukraine (например) — Синее if (path.includes('/ukraine/')) { return { color: "#0000FF", weight: 2, opacity: 0.8 }; } // Остальное — Белое или любое другое return { color: "#FFFFFF", weight: 2, opacity: 0.7 }; } // 3. Удаление инструментов рисования и экспорта // Просто не вызывайте L.Control.Draw или удалите эти строки: // map.removeControl(drawControl); // $('.leaflet-draw').hide(); // Если используете jQuery // 4. Загрузка слоев с фильтром по дате var allGeoJSONLayers = []; function loadLayer(url) { fetch(url) .then(response => response.json()) .then(data => { var layer = L.geoJSON(data, { style: function(feature) { return getLayerStyle(url, feature); }, filter: function(feature) { // Логика фильтрации по дате (если в свойствах объекта есть 'date') if (!currentDate) return true; return new Date(feature.properties.date) <= new Date(currentDate); } }).addTo(map); allGeoJSONLayers.push({layer: layer, data: data, url: url}); }); } // 5. Временная шкала (Timeline) var slider = document.getElementById('dateSlider'); slider.oninput = function() { var selectedDate = this.value; // Например, "2023-12-01" updateMapByDate(selectedDate); }; function updateMapByDate(date) { allGeoJSONLayers.forEach(obj => { map.removeLayer(obj.layer); // Убираем старый слой obj.layer = L.geoJSON(obj.data, { style: function(feature) { return getLayerStyle(obj.url, feature); }, filter: function(feature) { // Показываем только те объекты, которые появились ДО выбранной даты return new Date(feature.properties.date) <= new Date(date); } }).addTo(map); }); }
